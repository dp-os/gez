HTML#

이 튜토리얼은 Gez 기반의 HTML SSR 애플리케이션을 처음부터 구축하는 방법을 안내합니다. Gez 프레임워크를 사용하여 서버 사이드 렌더링 애플리케이션을 만드는 방법을 완전한 예제를 통해 보여드리겠습니다.

프로젝트 구조#

먼저, 프로젝트의 기본 구조를 살펴보겠습니다:

1.
2├── package.json         # 프로젝트 설정 파일, 의존성 및 스크립트 명령어 정의
3├── tsconfig.json        # TypeScript 설정 파일, 컴파일 옵션 설정
4└── src                  # 소스 코드 디렉토리
5    ├── app.ts           # 메인 애플리케이션 컴포넌트, 페이지 구조 및 상호작용 로직 정의
6    ├── create-app.ts    # 애플리케이션 인스턴스 생성 팩토리, 애플리케이션 초기화 담당
7    ├── entry.client.ts  # 클라이언트 진입 파일, 브라우저 측 렌더링 처리
8    ├── entry.node.ts    # Node.js 서버 진입 파일, 개발 환경 설정 및 서버 시작 담당
9    └── entry.server.ts  # 서버 진입 파일, SSR 렌더링 로직 처리

프로젝트 설정#

package.json#

package.json 파일을 생성하여 프로젝트 의존성 및 스크립트를 설정합니다:

package.json
1{
2  "name": "ssr-demo-html",
3  "version": "1.0.0",
4  "type": "module",
5  "private": true,
6  "scripts": {
7    "dev": "gez dev",
8    "build": "npm run build:dts && npm run build:ssr",
9    "build:ssr": "gez build",
10    "preview": "gez preview",
11    "start": "NODE_ENV=production node dist/index.js",
12    "build:dts": "tsc --declaration --emitDeclarationOnly --outDir dist/src"
13  },
14  "dependencies": {
15    "@gez/core": "*"
16  },
17  "devDependencies": {
18    "@gez/rspack": "*",
19    "@types/node": "22.8.6",
20    "typescript": "^5.7.3"
21  }
22}

package.json 파일을 생성한 후, 프로젝트 의존성을 설치해야 합니다. 다음 명령어 중 하나를 사용하여 설치할 수 있습니다:

1pnpm install
2# 또는
3yarn install
4# 또는
5npm install

이 명령어는 TypeScript 및 SSR 관련 의존성을 포함한 모든 필수 패키지를 설치합니다.

소스 코드 구조#

app.ts#

메인 애플리케이션 컴포넌트 src/app.ts를 생성하여 페이지 구조 및 상호작용 로직을 구현합니다:

src/app.ts
1/**
2 * @file 예제 컴포넌트
3 * @description Gez 프레임워크의 기본 기능을 보여주기 위해 자동으로 업데이트되는 시간을 표시하는 페이지 제목을 보여줍니다.
4 */
5
6export default class App {
7    /**
8     * 현재 시간, ISO 형식 사용
9     * @type {string}
10     */
11    public time = '';
12
13    /**
14     * 애플리케이션 인스턴스 생성
15     * @param {SsrContext} [ssrContext] - 서버 사이드 컨텍스트, 가져오기 메타데이터 집합 포함
16     */
17    public constructor(public ssrContext?: SsrContext) {
18        // 생성자에서 추가 초기화가 필요하지 않음
19    }
20
21    /**
22     * 페이지 내용 렌더링
23     * @returns {string} 페이지 HTML 구조 반환
24     */
25    public render(): string {
26        // 서버 사이드 환경에서 가져오기 메타데이터를 올바르게 수집
27        if (this.ssrContext) {
28            this.ssrContext.importMetaSet.add(import.meta);
29        }
30
31        return `
32        <div id="app">
33            <h1><a href="https://www.jsesm.com/guide/frameworks/html.html" target="_blank">Gez 빠른 시작</a></h1>
34            <time datetime="${this.time}">${this.time}</time>
35        </div>
36        `;
37    }
38
39    /**
40     * 클라이언트 초기화
41     * @throws {Error} 시간 표시 요소를 찾을 수 없을 때 오류 발생
42     */
43    public onClient(): void {
44        // 시간 표시 요소 가져오기
45        const time = document.querySelector('#app time');
46        if (!time) {
47            throw new Error('시간 표시 요소를 찾을 수 없습니다');
48        }
49
50        // 1초마다 시간 업데이트
51        setInterval(() => {
52            this.time = new Date().toISOString();
53            time.setAttribute('datetime', this.time);
54            time.textContent = this.time;
55        }, 1000);
56    }
57
58    /**
59     * 서버 초기화
60     */
61    public onServer(): void {
62        this.time = new Date().toISOString();
63    }
64}
65
66/**
67 * 서버 사이드 컨텍스트 인터페이스
68 * @interface
69 */
70export interface SsrContext {
71    /**
72     * 가져오기 메타데이터 집합
73     * @type {Set<ImportMeta>}
74     */
75    importMetaSet: Set<ImportMeta>;
76}

entry.node.ts#

entry.node.ts 파일을 생성하여 개발 환경 설정 및 서버 시작을 구성합니다:

src/entry.node.ts
1/**
2 * @file Node.js 서버 진입 파일
3 * @description 개발 환경 설정 및 서버 시작 담당, SSR 런타임 환경 제공
4 */
5
6import http from 'node:http';
7import type { GezOptions } from '@gez/core';
8
9export default {
10    /**
11     * 개발 환경 애플리케이션 생성기 구성
12     * @description Rspack 애플리케이션 인스턴스 생성 및 구성, 개발 환경 빌드 및 핫 리로드 지원
13     * @param gez Gez 프레임워크 인스턴스, 핵심 기능 및 설정 인터페이스 제공
14     * @returns HMR 및 실시간 미리보기를 지원하는 구성된 Rspack 애플리케이션 인스턴스 반환
15     */
16    async devApp(gez) {
17        return import('@gez/rspack').then((m) =>
18            m.createRspackHtmlApp(gez, {
19                config(context) {
20                    // 여기서 Rspack 컴파일 설정을 사용자 정의
21                }
22            })
23        );
24    },
25
26    /**
27     * HTTP 서버 구성 및 시작
28     * @description HTTP 서버 인스턴스 생성, Gez 미들웨어 통합, SSR 요청 처리
29     * @param gez Gez 프레임워크 인스턴스, 미들웨어 및 렌더링 기능 제공
30     */
31    async server(gez) {
32        const server = http.createServer((req, res) => {
33            // Gez 미들웨어를 사용하여 요청 처리
34            gez.middleware(req, res, async () => {
35                // 서버 사이드 렌더링 실행
36                const rc = await gez.render({
37                    params: { url: req.url }
38                });
39                res.end(rc.html);
40            });
41        });
42
43        server.listen(3000, () => {
44            console.log('서버 시작: http://localhost:3000');
45        });
46    }
47} satisfies GezOptions;

이 파일은 개발 환경 설정 및 서버 시작을 위한 진입 파일로, 두 가지 핵심 기능을 포함합니다:

  1. devApp 함수: 개발 환경의 Rspack 애플리케이션 인스턴스를 생성 및 구성하며, 핫 리로드 및 실시간 미리보기 기능을 지원합니다.
  2. server 함수: HTTP 서버를 생성 및 구성하고, Gez 미들웨어를 통합하여 SSR 요청을 처리합니다.

entry.server.ts#

서버 사이드 렌더링 진입 파일 src/entry.server.ts를 생성합니다:

src/entry.server.ts
1/**
2 * @file 서버 사이드 렌더링 진입 파일
3 * @description 서버 사이드 렌더링 프로세스, HTML 생성 및 리소스 주입 담당
4 */
5
6import type { RenderContext } from '@gez/core';
7import type App from './app';
8import type { SsrContext } from './app';
9import { createApp } from './create-app';
10
11// 페이지 내용 생성 로직 캡슐화
12const renderToString = (app: App, ssrContext: SsrContext): string => {
13    // 서버 사이드 렌더링 컨텍스트를 애플리케이션 인스턴스에 주입
14    app.ssrContext = ssrContext;
15    // 서버 초기화
16    app.onServer();
17
18    // 페이지 내용 생성
19    return app.render();
20};
21
22export default async (rc: RenderContext) => {
23    // 애플리케이션 인스턴스 생성, app 인스턴스를 포함한 객체 반환
24    const { app } = createApp();
25    // renderToString을 사용하여 페이지 내용 생성
26    const html = renderToString(app, {
27        importMetaSet: rc.importMetaSet
28    });
29
30    // 의존성 수집 제출, 모든 필수 리소스가 로드되도록 보장
31    await rc.commit();
32
33    // 완전한 HTML 구조 생성
34    rc.html = `<!DOCTYPE html>
35<html lang="ko">
36<head>
37    ${rc.preload()}
38    <title>Gez 빠른 시작</title>
39    ${rc.css()}
40</head>
41<body>
42    ${html}
43    ${rc.importmap()}
44    ${rc.moduleEntry()}
45    ${rc.modulePreload()}
46</body>
47</html>
48`;
49};

프로젝트 실행#

위의 파일 설정을 완료한 후, 다음 명령어를 사용하여 프로젝트를 실행할 수 있습니다:

  1. 개발 모드:
1npm run dev
  1. 프로젝트 빌드:
1npm run build
  1. 프로덕션 환경 실행:
1npm run start

이제 Gez 기반의 HTML SSR 애플리케이션을 성공적으로 생성했습니다! http://localhost:3000 에 접속하여 결과를 확인할 수 있습니다.
